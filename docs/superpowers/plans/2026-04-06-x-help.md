# x-help Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create `x-help`, a dotfiles documentation browser with interactive fzf selection and direct name lookup.

**Architecture:** Single bash script in `local/bin/x-help` with a `discover_files` function for `find`-based `.md` discovery, a `direct_lookup` function for name-based matching, and a `main` function that dispatches between interactive and direct modes. Companion docs in `local/bin/x-help.md`.

**Tech Stack:** Bash, fzf, find, msgr, bats (testing)

### Task 2: Create the script with discovery function

- [ ] **Step 1: Create x-help with shebang, config, msgr, and discover_files**

```bash
#!/usr/bin/env bash
# x-help — Browse and display dotfiles documentation
# @description Browse and display dotfiles documentation

set -euo pipefail

DOTFILES="${DOTFILES:-$HOME/.dotfiles}"

# Source msgr for colored output
MSGR="${DOTFILES}/local/bin/msgr"
if [[ -f "$MSGR" ]]; then
  # shellcheck source=msgr
  source "$MSGR"
fi

# Allow sourcing for tests without running main
if [[ "${1:-}" == "--source-only" ]]; then
  # Functions are loaded, skip main execution
  _X_HELP_SOURCE_ONLY=1
fi

# Discover all .md documentation files in the dotfiles tree.
# Outputs relative paths (e.g., local/bin/x-foreach.md), sorted.
discover_files()
{
  find "$DOTFILES" -name '*.md' -type f \
    -not -path '*/tools/*' \
    -not -path '*/docs/plans/*' \
    -not -path '*/docs/superpowers/*' \
    -not -path '*/config/tmux/plugins/*' \
    -not -path '*/node_modules/*' \
    -not -path '*/.git/*' \
    | sed "s|^${DOTFILES}/||" \
    | sort -u
}

main()
{
  :
}

if [[ "${_X_HELP_SOURCE_ONLY:-0}" != "1" ]]; then
  main "$@"
fi
```

### Task 4: Implement direct mode lookup

**Files:**
- Modify: `local/bin/x-help`

- [ ] **Step 1: Add the direct_lookup function**

Add after the `discover_files` function, before `main`:

```bash
# Look up documentation by bare name (without .md extension).
# Returns: exit 0 + prints path if exactly one match,
#          exit 2 + prints paths if multiple matches,
#          exit 1 if no match found.
# $1 - bare name to search for (e.g., "x-foreach")
direct_lookup()
{
  local name="$1"
  local target="${name}.md"
  local matches

  matches=$(discover_files | while IFS= read -r path; do
    if [[ "$(basename "$path")" == "$target" ]]; then
      echo "$path"
    fi
  done)

  if [[ -z "$matches" ]]; then
    return 1
  fi

  local count
  count=$(echo "$matches" | wc -l)

  if [[ "$count" -eq 1 ]]; then
    echo "$matches"
    return 0
  else
    echo "$matches"
    return 2
  fi
}
```

Replace the `main` function:

```bash
main()
{
  # Check for fzf
  if ! command -v fzf > /dev/null 2>&1; then
    msgr err "fzf is required but not installed"
    exit 1
  fi

  # Direct mode: x-help <name>
  if [[ $# -ge 1 ]]; then
    local name="$1"
    local result
    local exit_code=0
    result=$(direct_lookup "$name") || exit_code=$?

    case "$exit_code" in
      0)
        # Single match — display it
        cat "$DOTFILES/$result"
        ;;
      2)
        # Multiple matches — fzf to disambiguate
        local selected
        selected=$(echo "$result" | fzf) || exit 0
        [[ -z "$selected" ]] && exit 0
        cat "$DOTFILES/$selected"
        ;;
      *)
        # No match
        msgr err "No documentation found for '$name'"
        exit 1
        ;;
    esac
    return
  fi

  # Interactive mode: x-help
  local files
  files=$(discover_files)

  if [[ -z "$files" ]]; then
    msgr warn "No documentation files found"
    exit 0
  fi

  local selected
  selected=$(echo "$files" | fzf) || exit 0
  [[ -z "$selected" ]] && exit 0
  cat "$DOTFILES/$selected"
}
```
